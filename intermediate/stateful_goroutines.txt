package main

import (
	"fmt"
	"time"
)

type StatefulGoroutine struct {
	count int
	ch    chan int
}

func (w *StatefulGoroutine) start() {
	go func() {
		for {
			select {
			case value := <-w.ch:
				w.count += value
				fmt.Println(w.count)
			}
		}
	}()
}

func (w *StatefulGoroutine) send(value int) {
	w.ch <- value
}

func main() {

	stWorker := StatefulGoroutine{
		ch: make(chan int),
	}
	stWorker.start()

	for i := range 5 {
		stWorker.send(i)
		time.Sleep(500 * time.Millisecond)
	}
}
