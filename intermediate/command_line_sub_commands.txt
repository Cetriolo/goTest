package main

import (
	"flag"
	"fmt"
	"os"
)

func main() {

	stringFlag := flag.String("user", "default value", "a string flag")
	flag.Parse()
	fmt.Println("user:", *stringFlag)

	
	subcommand1 := flag.NewFlagSet("firstSub", flag.ExitOnError)

	subcommand2 := flag.NewFlagSet("secondSub", flag.ExitOnError)

	firstFlag := subcommand1.Bool("processing", false, "Command processing")

	secondFlag := subcommand2.Int("bytes", 1024, "Number of bytes")

	flagsc2 := subcommand2.String("language", "GO", "Enter your language")

	if len(os.Args) < 2 {
		fmt.Println("expected 'firstSub' or 'secondSub' subcommands")
		os.Exit(1)
	}
	switch os.Args[1] {
	case "firstSub":
		subcommand1.Parse(os.Args[2:])
		fmt.Println("subcommand1:")
		fmt.Println("  processing:", *firstFlag)
		fmt.Println()

	case "secondSub":
		subcommand2.Parse(os.Args[2:])
		fmt.Println("subcommand2:")
		fmt.Println("  bytes:", *secondFlag)
		fmt.Println("  language:", *flagsc2)
	default:
		fmt.Println("expected 'firstSub' or 'secondSub' subcommands")
		os.Exit(1)
	}
}
