package main

import (
	"errors"
	"fmt"
)

func main() {
	err := doSomething()
	if err != nil {
		fmt.Println("Error:", err)
		return
	}
	fmt.Println("OK")
}

type customError struct {
	code    int
	message string
	err     error
}

func (e *customError) Error() string {
	return fmt.Sprintf("customError %d: %s, %v", e.code, e.message, e.err)
}
