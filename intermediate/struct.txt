package main

import (
	"fmt"
)

func main() {

	p := Person{
		firstName: "John",
		lastName:  "Doe",
		age:       30,
	}

	fmt.Println(p)

	p1 := Person{
		firstName: "Jane",
		age:       25,
		address: Address{
			city:  "Los Angeles",
			state: "CA",
		},
		PhoneHomeCell: PhoneHomeCell{
			home: "555-5678",
			cell: "555-1234",
		},
	}
	p.address.city = "New York"
	p.address.state = "NY"
	fmt.Println(p1)

	fmt.Println(p1.firstName)

	user := struct {
		username string
		email    string
	}{
		username: "jane_doe",
		email:    "jane@example.com",
	}

	fmt.Println(user)
	p1.incrementAgeBy1()
	fmt.Println(p1)

	p1.cell = "555-1234"
	p1.home = "555-5678"
}

type Person struct {
	firstName string
	lastName  string
	age       int
	address   Address
	PhoneHomeCell
}

func (p Person) fullName() string {
	return p.firstName + " " + p.lastName
}

func (p *Person) incrementAgeBy1() {
	p.age++
}

type Address struct {
	city  string
	state string
}

type PhoneHomeCell struct {
	home string
	cell string
}
