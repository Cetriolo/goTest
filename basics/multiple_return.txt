package main

import (
	"errors"
	"fmt"
)

func main() {
	//func functionName(p1 type1, p2 type2) (returnType1, returnType2) {
	//	// function body
	//}

	a, b := divide(10, 3)
	fmt.Println("Quotient:", a)
	fmt.Println("Modulus:", b)

	result, err := compare(3, 3)
	if err != nil {
		fmt.Println("Error:", err)
	} else {
		fmt.Println("Comparison Result:", result)
	}
}

func divide(a, b int) (quotient int, modulus int) {
	quotient = a / b
	modulus = a % b
	return
}

func compare(a, b int) (string, error) {
	if a > b {
		return "a is greater than b", nil
	} else if b > a {
		return "b is greater than a", nil
	} else {
		return "", errors.New("Unable to compare which is greater")
	}
}
